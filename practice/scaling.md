class MultiLevelCache:
    def __init__(self):
        # L1: In-memory cache (—Å–∞–º—ã–π –±—ã—Å—Ç—Ä—ã–π)
        self.l1_cache = {}

    def get_cache_key(self, func_name, *args, **kwargs):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∫–ª—é—á–∞ –∫—ç—à–∞"""
        key_data = {
            'func': func_name,
            'args': args,
            'kwargs': kwargs
        }
        return hashlib.md5(json.dumps(key_data, sort_keys=True).encode()).hexdigest()
    
    def get(self, key):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –∏–∑ –∫—ç—à–∞"""
        # L1: In-memory cache

            return self.l1_cache[key]
        
        # L2: Redis cache
        try:
            value = self.redis_client.get(key)
            if value:
                # –ü–æ–¥–Ω–∏–º–∞–µ–º –≤ L1
                self.l1_cache[key] = json.loads(value)
                return self.l1_cache[key]
        except Exception as e:
            logger.warning(f"Redis cache error: {e}")
        
        # L3: Database

            value = self.db_client.get(key)
            if value:
                # –ü–æ–¥–Ω–∏–º–∞–µ–º –≤ L2 –∏ L1
                self.redis_client.setex(key, 3600, json.dumps(value))  # 1 —á–∞—Å
                self.l1_cache[key] = value
                return value
        except Exception as e:
            logger.error(f"Database error: {e}")
        
        return None
    
    def set(self, key, value, ttl=3600):
        """–°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –≤ –∫—ç—à"""

        self.l1_cache[key] = value
        
        # L2: Redis cache

            self.redis_client.setex(key, ttl, json.dumps(value))
        except Exception as e:
            logger.warning(f"Redis cache error: {e}")
        
        # L3: Database (–¥–ª—è –∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏ –≤–∞–∂–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö)
        if ttl > 86400:  # –ë–æ–ª—å—à–µ —Å—É—Ç–æ–∫

                self.db_client.set(key, value)
            except Exception as e:
                logger.error(f"Database error: {e}")

def cached(ttl=3600):

        cache = MultiLevelCache()
        
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_key = cache.get_cache_key(func.__name__, *args, **kwargs)
            
            # –ü–æ–ø—ã—Ç–∫–∞ –ø–æ–ª—É—á–∏—Ç—å –∏–∑ –∫—ç—à–∞
            cached_result = cache.get(cache_key)
            if cached_result is not None:
                return cached_result
            
            # –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏
            result = func(*args, **kwargs)
            
            # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –≤ –∫—ç—à
            cache.set(cache_key, result, ttl)
            
            return result
        
        return wrapper
    return decorator
